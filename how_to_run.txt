# RogueShield "earlier model" workflow (Attacks Training.ipynb)
# This keeps the original multiclass model path:
#   intrusion_classifier.keras + preprocessor.joblib + label_encoder.joblib

# 0) Stop all running capture/classifier processes (including DoS-only detector if it was started)
Get-CimInstance Win32_Process |
  Where-Object {
    $_.Name -eq 'dumpcap.exe' -or
    $_.Name -eq 'tshark.exe' -or
    ($_.Name -eq 'python.exe' -and (
      $_.CommandLine -match 'live_classifier_stream.py' -or
      $_.CommandLine -match 'dos_live_detector.py' -or
      $_.CommandLine -match 'live_monitor_runner.py'
    ))
  } |
  ForEach-Object { Stop-Process -Id $_.ProcessId -Force }

# Optional: stop iperf generators/servers
Get-CimInstance Win32_Process |
  Where-Object { $_.Name -eq 'iperf3.exe' } |
  ForEach-Object { Stop-Process -Id $_.ProcessId -Force }


# 1) Train/retrain the base earlier model in notebook
# Open and run all cells in:
#   C:\Users\anike\Downloads\Attacks Training.ipynb
# Expected outputs:
#   C:\Users\anike\Downloads\Attack datasets\rogueshield_outputs\intrusion_classifier.keras
#   C:\Users\anike\Downloads\Attack datasets\rogueshield_outputs\preprocessor.joblib
#   C:\Users\anike\Downloads\Attack datasets\rogueshield_outputs\label_encoder.joblib


# 2) Clean old live capture/log files
$capDir = "C:\Users\anike\Downloads\captures"
New-Item -ItemType Directory -Force -Path $capDir | Out-Null

Remove-Item "$capDir\live_*.pcapng" -Force -ErrorAction SilentlyContinue
Remove-Item "$capDir\classification_live.log" -Force -ErrorAction SilentlyContinue
Remove-Item "$capDir\classification_live.csv" -Force -ErrorAction SilentlyContinue
Remove-Item "$capDir\classifier_proc.out.log" -Force -ErrorAction SilentlyContinue
Remove-Item "$capDir\classifier_proc.err.log" -Force -ErrorAction SilentlyContinue
Remove-Item "$capDir\dumpcap_live.log" -Force -ErrorAction SilentlyContinue


# 3) Start fresh live capture
# Interface example: 5 (Wi-Fi). Use "dumpcap -D" to list interfaces on your machine.
$cap = Start-Process -FilePath "C:\Program Files\Wireshark\dumpcap.exe" `
  -ArgumentList "-i","5","-b","duration:15","-b","files:200","-w","$capDir\live.pcapng" `
  -RedirectStandardError "$capDir\dumpcap_live.log" `
  -WindowStyle Hidden -PassThru
$cap.Id


# 4) Start earlier live classifier (uses intrusion_classifier*.keras)
$cls = Start-Process -FilePath "C:\Users\anike\AppData\Local\Programs\Python\Python311\python.exe" `
  -ArgumentList "-u","C:\Users\anike\Downloads\live_classifier_stream.py" `
  -RedirectStandardOutput "$capDir\classifier_proc.out.log" `
  -RedirectStandardError "$capDir\classifier_proc.err.log" `
  -WindowStyle Hidden -PassThru
$cls.Id


# 5) Collect normal traffic for a few minutes, then fine-tune with your live captures
# Note: attack_alerts=0 lines are expected for normal traffic and are used as safe files.
& "C:\Users\anike\AppData\Local\Programs\Python\Python311\python.exe" `
  "C:\Users\anike\Downloads\retrain_with_live_normal.py" `
  --captures-glob "C:\Users\anike\Downloads\captures\live_*.pcapng" `
  --classification-log "C:\Users\anike\Downloads\captures\classification_live.log" `
  --epochs 4 --batch-size 256 --live-weight 8


# 6) Restart classifier so it loads intrusion_classifier_live_tuned.keras
Get-CimInstance Win32_Process |
  Where-Object { $_.Name -eq 'python.exe' -and $_.CommandLine -match 'live_classifier_stream.py' } |
  ForEach-Object { Stop-Process -Id $_.ProcessId -Force }

$cls = Start-Process -FilePath "C:\Users\anike\AppData\Local\Programs\Python\Python311\python.exe" `
  -ArgumentList "-u","C:\Users\anike\Downloads\live_classifier_stream.py" `
  -RedirectStandardOutput "$capDir\classifier_proc.out.log" `
  -RedirectStandardError "$capDir\classifier_proc.err.log" `
  -WindowStyle Hidden -PassThru
$cls.Id


# 7) Monitor
Get-Content "C:\Users\anike\Downloads\captures\classification_live.log" -Wait -Tail 30
Get-Content "C:\Users\anike\Downloads\captures\dumpcap_live.log" -Wait -Tail 30
